#############################################
##### Testing
#############################################

set(prefix "${MODULE}")

add_test(
    NAME ${prefix}_can_Run
    COMMAND ${APPLICATION}
)
set_tests_properties (${prefix}_can_Run PROPERTIES
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Validation_incorrect_input
    COMMAND ${APPLICATION} sdsfp 02 asd5
)
set_tests_properties (${prefix}_Validation_incorrect_input PROPERTIES
    PASS_REGULAR_EXPRESSION "Incorrect input."
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Validate_less_arguments
    COMMAND ${APPLICATION} sp
)
set_tests_properties (${prefix}_Validate_less_arguments PROPERTIES
    PASS_REGULAR_EXPRESSION "Incorrect input."
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Validation_many_arguments
    COMMAND ${APPLICATION} sp 02 asd5 2 32 13
)
set_tests_properties (${prefix}_Validation_many_arguments PROPERTIES
    PASS_REGULAR_EXPRESSION "Incorrect input."
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Validation_when_one_vertex_is_incorrect
    COMMAND ${APPLICATION} sp 1 -7
)
set_tests_properties (${prefix}_Validation_when_one_vertex_is_incorrect PROPERTIES
    PASS_REGULAR_EXPRESSION "Error with argument 2: not a number"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Validation_help_information
    COMMAND ${APPLICATION} help
)
set_tests_properties (${prefix}_Validation_help_information PROPERTIES
    PASS_REGULAR_EXPRESSION "Dijckstra's Algorithm.\n"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Shortest_path_between_two_vertex_which_equals
    COMMAND ${APPLICATION} sp 2 2
)
set_tests_properties (${prefix}_Shortest_path_between_two_vertex_which_equals PROPERTIES
    PASS_REGULAR_EXPRESSION "0"
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Shortest_path_between_two_vertex_front_direction
    COMMAND ${APPLICATION} sp 5 0
)
set_tests_properties (${prefix}_Shortest_path_between_two_vertex_front_direction PROPERTIES
    PASS_REGULAR_EXPRESSION "1, 3, 6, 5"
    TIMEOUT 900
    LABELS "${MODULE}")

add_test(
    NAME ${prefix}_Shortest_path_between_two_vertex_reverse_direction
    COMMAND ${APPLICATION} sp 0 5
)
set_tests_properties (${prefix}_Shortest_path_between_two_vertex_reverse_direction PROPERTIES
    PASS_REGULAR_EXPRESSION "1, 3, 6, 5"
    TIMEOUT 900
    LABELS "${MODULE}")


add_test(
    NAME ${prefix}_Cant_do_work_without_vertexes
    COMMAND ${APPLICATION} init graph 0
)
set_tests_properties (${prefix}_Cant_do_work_without_vertexes PROPERTIES
    PASS_REGULAR_EXPRESSION "Error: graph has no vertexes"
    LABELS "${MODULE}")
